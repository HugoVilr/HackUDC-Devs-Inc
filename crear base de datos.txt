
CREACION  TABLA

-- Tabla de usuarios
CREATE TABLE usuarios (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    telefono VARCHAR(20),
    departamento VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Tabla de competencias
CREATE TABLE competencias (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL UNIQUE,
    descripcion TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Tabla intermedia para relacionar usuarios y competencias
CREATE TABLE usuario_competencia (
    usuario_id INT NOT NULL,
    competencia_id INT NOT NULL,
    nivel VARCHAR(50),        -- Ejemplo: 'Básico', 'Intermedio', 'Avanzado'
    certificacion VARCHAR(100), -- Detalle de certificaciones o títulos
    PRIMARY KEY (usuario_id, competencia_id),
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE,
    FOREIGN KEY (competencia_id) REFERENCES competencias(id) ON DELETE CASCADE
);
